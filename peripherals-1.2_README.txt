require "utils.vship" :

    [[returns the directions of the vs ship]]
    .getLook() -> look vector
    .getUp() -> up vector
    .getRight() -> right vector

    [[returns the angles of the ship without a rotation order]]
    .getVectorYaw() -> yaw
    .getVectorPitch() -> pitch

    .getAngularVelocity() -> {yaw=.., pitch=.., roll=..}

    .transformToWorld(shipyard pos) -> world pos
    .transformToShip(world pos)  -> shipyard pos

    .getPositionAABB(shipyard min, shipyard max) -> world position
    .getCenterOfMass()

    .blockChanged() -> world position, shipyard position, change of mass
